## Note to AI agent

This plan is detailed by design. Where it specifies file paths, method names, migration column lists, tests, that's not because they're clever — it's because ambiguity is how you produce bugs in a long agent session. Follow the letter. Where the plan asks you to make a judgment call, make it and note what you chose in the commit message.

**Forbidden**:
- No service objects (`app/services/` is off limits).
- No form objects, presenter objects, or "policy" objects.
- No new gems beyond what's already in the Gemfile.
- No premature extraction of helpers, concerns, or shared modules until at least three duplicates exist.
- No defensive code for cases that can't happen.
- No rescuing exceptions you don't have a specific recovery for.
- Multi-row write logic lives on the relevant Active Record model as fat-model methods, not in service-object classes.

### W1.1 Family-form conditional reveal bug

The bug is in `app/javascript/controllers/family_form_controller.js`. The conditional reveals don't fire on `connect()` or on radio change.

**Investigate first**: open the browser, visit `/mobile/onboarding/family`, open DevTools, change relationship to "Married," watch the console for errors. Most likely cause: `data-action` wiring is incorrect, or the controller's `connect()` doesn't read the initial radio state.

**Fix**: ensure the controller has a `refreshConditionals()` method that runs on both `connect()` and on every radio change:

```javascript
import { Controller } from "@hotwired/stimulus"

export default class extends Controller {
  static targets = ["relationshipKind", "partnerFields", "hasChildren", "childrenSection"]

  connect() {
    this.refreshConditionals()
  }

  onRelationshipChange() {
    this.refreshConditionals()
  }

  onHasChildrenChange() {
    this.refreshConditionals()
  }

  refreshConditionals() {
    const rel = this.relationshipKindTarget.querySelector("input:checked")?.value
    const hasPartner = ["spouse", "civil-partnership", "married", "cohabiting"].includes(rel)
    this.partnerFieldsTarget.hidden = !hasPartner

    const hasChildren = this.hasChildrenTarget.querySelector("input:checked")?.value === "yes"
    this.childrenSectionTarget.hidden = !hasChildren
  }
}
```

The view (`app/views/mobile/onboarding/family.html.erb`) needs `data-action="change->family-form#onRelationshipChange"` on the relationship_status radios and equivalent on has_children radios.

**Done when**: visiting `/mobile/onboarding/family`, picking "Married", reveals partner fields. Picking "Yes" under children, reveals children section. Reload after picking — state persists and conditional reveals match.

### W2.3 Data migration: existing OnboardingSession data → Person + Marriage + Parentage

```ruby
class MigrateOnboardingDataToRecords < ActiveRecord::Migration[8.1]
  disable_ddl_transaction!  # Active Record encryption needs an active connection

  def up
    OnboardingSession.find_each do |session|
      ActiveRecord::Base.transaction do
        will_maker = create_will_maker_for(session) if session.first_name.present?
        next if will_maker.nil?

        partner = create_partner_for(session, will_maker) if session.spouse_first_name.present?
        create_children_for(session, will_maker, partner)
      end
    end
  end

  def down
    Person.where.not(onboarding_session_id: nil).destroy_all
  end

  private

  def create_will_maker_for(session)
    Person.create!(
      onboarding_session: session,
      relationship_kind: "self",
      first_name: session.first_name,
      middle_names: session.middle_names,
      last_name: session.last_name,
      date_of_birth: session.date_of_birth,
      country_of_residence: session.country_of_residence,
      nationality: session.nationality,
      domiciled_in_uk: session.domiciled_in_uk,
      currently_resident_in_uk: session.currently_resident_in_uk,
      times_divorced: session.divorce_status == "yes" ? 1 : 0,
      parents_alive: session.parents_alive,
      parents_in_law_alive: session.parents_in_law_alive,
      siblings_alive: session.siblings_alive,
      number_of_siblings: session.number_of_siblings
    )
  end

  def create_partner_for(session, will_maker)
    partner = Person.create!(
      onboarding_session: session,
      relationship_kind: "spouse",
      first_name: session.spouse_first_name,
      last_name: session.spouse_last_name
    )
    Marriage.create!(
      will_maker_person: will_maker,
      partner_person: partner,
      kind: marriage_kind_from_legacy(session.relationship_status),
      phase: "active",
      started_at: nil  # legacy data has no date; let user fill in next visit
    )
    partner
  end

  def create_children_for(session, will_maker, partner)
    return if session.children_payload.blank?

    Array(session.children_payload).each_with_index do |child_data, idx|
      child = Person.create!(
        onboarding_session: session,
        relationship_kind: "child",
        position: idx,
        first_name: child_data["first_name"],
        last_name: child_data["last_name"],
        date_of_birth: child_data["date_of_birth"],
        disabled: child_data["disabled"] == "yes",
        lacks_mental_capacity: child_data["lacks_mental_capacity"] == "yes"
      )
      Parentage.create!(
        parent_person: will_maker,
        child_person: child,
        kind: child_data["relationship"] || "biological"
      )
    end
  end

  def marriage_kind_from_legacy(relationship_status)
    case relationship_status
    when "married" then "married"
    when "civil-partnership", "civil_partnership" then "civil_partnership"
    when "cohabiting" then "cohabiting"
    else "married"
    end
  end
end
```

### W2.5 Refactor controller + add fat-model methods on OnboardingSession

`app/models/onboarding_session.rb` gets the multi-row write logic as fat-model methods (Elon R1):

```ruby
class OnboardingSession < ApplicationRecord
  has_many :people, dependent: :destroy
  has_one :will_maker_person, -> { where(relationship_kind: "self") },
          class_name: "Person", inverse_of: :onboarding_session
  has_one :partner_person, -> { where(relationship_kind: "spouse") },
          class_name: "Person", inverse_of: :onboarding_session

  # === Welcome step ===
  def apply_welcome_step!(params)
    transaction do
      will_maker = will_maker_person ||
                   create_will_maker_person!(relationship_kind: "self")
      will_maker.assign_attributes(params)
      will_maker.save!(context: :welcome_step)
      update!(current_step: "location")
    end
  end

  # === Location step ===
  def apply_location_step!(params)
    transaction do
      will_maker_person.assign_attributes(params)
      will_maker_person.save!(context: :location_step)
      update!(current_step: "family")
    end
  end

  # === Family step ===
  def apply_family_step!(params)
    transaction do
      apply_partner!(params)
      apply_children!(params)
      update!(current_step: "extended_family")
    end
  end

  # === Extended family step ===
  def apply_extended_family_step!(params)
    transaction do
      will_maker_person.assign_attributes(params)
      will_maker_person.save!(context: :extended_family_step)
      update!(current_step: "wrap_up")
    end
  end

  private

  def apply_partner!(params)
    case params[:relationship_kind]
    when "spouse", "partner", "married", "civil_partnership", "cohabiting"
      partner = partner_person ||
                people.create!(relationship_kind: "spouse")
      partner.assign_attributes(params.require(:partner).permit(:first_name, :last_name))
      partner.save!(context: :family_step)

      marriage = Marriage.find_or_initialize_by(
        will_maker_person: will_maker_person,
        partner_person: partner
      )
      marriage.assign_attributes(
        kind: marriage_kind_from(params[:relationship_kind], params[:marriage_kind]),
        phase: "active",
        started_at: params.dig(:partner, :started_at)
      )
      marriage.save!
    when "single", "widowed"
      destroy_existing_partnership!
      will_maker_person.update!(times_widowed: params[:times_widowed].to_i) if params[:relationship_kind] == "widowed"
    end

    will_maker_person.update!(times_divorced: params[:times_divorced].to_i) if params[:times_divorced]
  end

  def apply_children!(params)
    if params[:has_children] != "yes"
      people.where(relationship_kind: "child").destroy_all
      destroy_orphaned_co_parents!
      return
    end

    submitted = Array(params[:children])
    submitted_positions = submitted.each_with_index.map { |_, i| i }
    people.where(relationship_kind: "child").where.not(position: submitted_positions).destroy_all

    submitted.each_with_index do |child_params, idx|
      child = people.find_or_initialize_by(relationship_kind: "child", position: idx)
      child.assign_attributes(
        child_params.permit(:first_name, :last_name, :date_of_birth, :disabled, :lacks_mental_capacity)
      )
      child.save!(context: :family_step)

      parentage = Parentage.find_or_initialize_by(
        parent_person: will_maker_person, child_person: child
      )
      parentage.kind = child_params[:relationship_kind]
      parentage.save!

      apply_co_parent!(child, child_params[:co_parent])
    end

    destroy_orphaned_co_parents!
  end

  def apply_co_parent!(child, co_parent_params)
    return if co_parent_params.blank?

    case co_parent_params[:type]
    when "partner"
      return if partner_person.nil?
      Parentage.find_or_create_by!(parent_person: partner_person, child_person: child) do |p|
        p.kind = co_parent_params[:relationship_to_child]
      end
    when "someone_else"
      co_parent = people.find_or_initialize_by(relationship_kind: "other", first_name: co_parent_params[:first_name], last_name: co_parent_params[:last_name])
      co_parent.save!(context: :family_step)
      Parentage.find_or_create_by!(parent_person: co_parent, child_person: child) do |p|
        p.kind = co_parent_params[:relationship_to_child]
      end
    when "deceased", "sole"
      # Nothing — user has sole PR
    end
  end

  def destroy_existing_partnership!
    partner = partner_person
    return if partner.nil?

    Parentage.where(parent_person: partner).destroy_all
    Marriage.where(will_maker_person: will_maker_person, partner_person: partner).destroy_all
    partner.destroy!
  end

  def destroy_orphaned_co_parents!
    people.where(relationship_kind: "other").includes(:parentages_as_parent).find_each do |co_parent|
      co_parent.destroy if co_parent.parentages_as_parent.empty?
    end
  end

  def marriage_kind_from(relationship_kind, marriage_kind_param)
    return marriage_kind_param if marriage_kind_param.present?
    case relationship_kind
    when "spouse", "married" then "married"
    when "civil_partnership" then "civil_partnership"
    when "cohabiting" then "cohabiting"
    end
  end
end
```

```ruby
# app/models/person.rb
class Person < ApplicationRecord
  # `relationship_kind = "self"` marks the will-maker Person.
  # No default scope (DHH dislikes them; explicit > implicit).
  # Consumers that surface "your people" should query .where.not(relationship_kind: "self")
  # OR use the helper class methods on the user (user.people_other_than_self).
  # Direct will-maker access: user.will_maker_person.

  belongs_to :user
  belongs_to :onboarding_session, optional: true
  has_many :parentages_as_parent, foreign_key: :parent_person_id, class_name: "Parentage", dependent: :destroy
  has_many :parentages_as_child, foreign_key: :child_person_id, class_name: "Parentage", dependent: :destroy
  has_many :marriages_as_will_maker, foreign_key: :will_maker_person_id, class_name: "Marriage", dependent: :destroy
  has_many :marriages_as_partner, foreign_key: :partner_person_id, class_name: "Marriage", dependent: :destroy

  validates :first_name, presence: true, on: %i[welcome_step family_step]
  validates :last_name, presence: true, on: %i[welcome_step family_step]
  validates :date_of_birth, presence: true, on: :welcome_step
  validate :age_within_bounds, on: :welcome_step

  validates :country_of_residence, presence: true, on: :location_step
  validates :nationality, presence: true, on: :location_step
  validates :domiciled_in_uk, presence: true, on: :location_step
  validates :currently_resident_in_uk, presence: true, on: :location_step

  validates :parents_alive, presence: true, on: :extended_family_step
  validates :siblings_alive, presence: true, on: :extended_family_step

  private

  def age_within_bounds
    return if date_of_birth.blank?
    age = ((Date.current - date_of_birth) / 365.25).to_i
    errors.add(:date_of_birth, "must indicate age 18 or older") if age < 18
    errors.add(:date_of_birth, "must indicate age 90 or younger") if age > 90
  end
end
```
